-- RANDOM TP & FOLLOW EVERY PLAYER
-- Follows each player for 2 seconds, then moves to a random next one.

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

local function getRandomizedPlayers()
    local list = {}
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer then
            table.insert(list, p)
        end
    end

    -- Shuffle
    for i = #list, 2, -1 do
        local j = math.random(1, i)
        list[i], list[j] = list[j], list[i]
    end

    return list
end


while task.wait() do
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local root = char:WaitForChild("HumanoidRootPart")

    local shuffled = getRandomizedPlayers()

    for _, target in ipairs(shuffled) do
        if target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
            local targetRoot = target.Character.HumanoidRootPart

            -- Teleport near them first
            root.CFrame = targetRoot.CFrame * CFrame.new(0, 0, 3)

            local followTime = 2
            local startTime = tick()

            -- FOLLOW LOOP (2 seconds)
            while tick() - startTime < followTime do
                if target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                    local tr = target.Character.HumanoidRootPart

                    -- Update your position to stay behind them
                    root.CFrame = tr.CFrame * CFrame.new(0, 0, 3)
                else
                    break
                end

                task.wait(0.05) -- smooth follow
            end
        end
    end
end
